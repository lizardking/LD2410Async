92LD2410Async::FrameReadResponse LD2410Async::readFrame()
93{
94 while (sensor->available()) {
95 byte b = sensor->read();
96
97 switch (readFrameState) {
98 case WAITING_FOR_HEADER:
99 if (b == LD2410Defs::headData[0]) {
100 readFrameHeaderIndex = 1;
101 readFrameState = DATA_HEADER;
102
103 }
104 else if (b == LD2410Defs::headConfig[0]) {
105 readFrameHeaderIndex = 1;
106 readFrameState = ACK_HEADER;
107
108 }
109 break;
110
111 case DATA_HEADER:
112 if (b == LD2410Defs::headData[readFrameHeaderIndex]) {
113 readFrameHeaderIndex++;
114
115 if (readFrameHeaderIndex == sizeof(LD2410Defs::headData)) {
116 receiveBufferIndex = 0;
117 readFrameState = READ_DATA_SIZE;
118 }
119 }
120 else if (b == LD2410Defs::headData[0]) {
121 // fallback: this byte might be the start of a new header
122
123 readFrameHeaderIndex = 1;
124
125 // stay in DATA_HEADER
126 }
127 else if (b == LD2410Defs::headConfig[0]) {
128 // possible start of config header
129 readFrameHeaderIndex = 1;
130 readFrameState = ACK_HEADER;
131
132 }
133 else {
134 // not a header at all
135 readFrameState = WAITING_FOR_HEADER;
136 readFrameHeaderIndex = 0;
137
138 }
139 break;
140
141 case ACK_HEADER:
142 if (b == LD2410Defs::headConfig[readFrameHeaderIndex]) {
143 readFrameHeaderIndex++;
144
145 if (readFrameHeaderIndex == sizeof(LD2410Defs::headConfig)) {
146 receiveBufferIndex = 0;
147 readFrameState = READ_ACK_SIZE;
148 }
149 }
150 else if (b == LD2410Defs::headConfig[0]) {
151 readFrameHeaderIndex = 1;
152 // stay in ACK_HEADER
153
154 }
155 else if (b == LD2410Defs::headData[0]) {
156 // maybe start of data header
157 readFrameHeaderIndex = 1;
158 readFrameState = DATA_HEADER;
159
160 }
161 else {
162 readFrameState = WAITING_FOR_HEADER;
163 readFrameHeaderIndex = 0;
164
165
166 }
167 break;
168
169 case READ_ACK_SIZE:
170 readFramePayloadSize(b, READ_ACK_PAYLOAD);
171 break;
172
173 case READ_DATA_SIZE:
174 readFramePayloadSize(b, READ_DATA_PAYLOAD);
175 break;
176
177 case READ_ACK_PAYLOAD:
178 return readFramePayload(b, LD2410Defs::tailConfig, ACK);
179
180 case READ_DATA_PAYLOAD:
181 return readFramePayload(b, LD2410Defs::tailData, DATA);
182
183 default:
184 readFrameState = WAITING_FOR_HEADER;
185 readFrameHeaderIndex = 0;
186 break;
187 }
188 }
189
190 return FAIL; // not enough bytes yet
191}

395bool LD2410Async::processData()
396{
397 DEBUG_PRINTBUF_DATA(receiveBuffer, receiveBufferIndex)
398
399 heartbeat();
400
401
402 if (((receiveBuffer[0] == 1) || (receiveBuffer[0] == 2)) && (receiveBuffer[1] == 0xAA))
403 {
404 configModeEnabled = false;
405
406 detectionData.timestamp = millis();
407 //Basic data
408 detectionData.targetState = LD2410Types::toTargetState(receiveBuffer[2] & 7);
409 switch (detectionData.targetState) {
410 case LD2410Types::TargetState::MOVING_TARGET:
411 detectionData.presenceDetected = true;
412 detectionData.movingPresenceDetected = true;
413 detectionData.stationaryPresenceDetected = false;
414 break;
415
416 case LD2410Types::TargetState::STATIONARY_TARGET:
417 detectionData.presenceDetected = true;
418 detectionData.movingPresenceDetected = false;
419 detectionData.stationaryPresenceDetected = true;
420 break;
421
422 case LD2410Types::TargetState::MOVING_AND_STATIONARY_TARGET:
423 detectionData.presenceDetected = true;
424 detectionData.movingPresenceDetected = true;
425 detectionData.stationaryPresenceDetected = true;
426 break;
427 default:
428 detectionData.presenceDetected = false;
429 detectionData.movingPresenceDetected = false;
430 detectionData.stationaryPresenceDetected = false;
431 break;
432 }
433 detectionData.movingTargetDistance = receiveBuffer[3] | (receiveBuffer[4] << 8);
434 detectionData.movingTargetSignal = receiveBuffer[5];
435 detectionData.stationaryTargetDistance = receiveBuffer[6] | (receiveBuffer[7] << 8);
436 detectionData.stationaryTargetSignal = receiveBuffer[8];
437 detectionData.detectedDistance = receiveBuffer[9] | (receiveBuffer[10] << 8);
438 detectionData.engineeringMode = receiveBuffer[0] == 1;
439
440 engineeringModeEnabled = detectionData.engineeringMode;
441
442 if (detectionData.engineeringMode)
443 { // Engineering mode data
444 detectionData.movingTargetGateSignalCount = receiveBuffer[11];
445 detectionData.stationaryTargetGateSignalCount = receiveBuffer[12];
446
447
448 int index = 13;
449 for (byte i = 0; i <= detectionData.movingTargetGateSignalCount; i++)
450 detectionData.movingTargetGateSignals[i] = receiveBuffer[index++];
451 for (byte i = 0; i <= detectionData.stationaryTargetGateSignalCount; i++)
452 detectionData.stationaryTargetGateSignals[i] = receiveBuffer[index++];
453
454 detectionData.outPinStatus = false;
455 detectionData.lightLevel = 0;
456 if (index < payloadSize) {
457 detectionData.lightLevel = receiveBuffer[index++];
458 if (index < payloadSize) {
459 detectionData.outPinStatus = receiveBuffer[index++];
460 }
461 }
462 }
463 else
464 { // Clear engineering mode data
465 detectionData.movingTargetGateSignalCount = 0;
466 detectionData.stationaryTargetGateSignalCount = 0;
467 detectionData.lightLevel = 0;
468 detectionData.outPinStatus = false;
469 }
470
471 if (rebootAsyncPending) {
472 rebootAsyncFinialize(LD2410Async::AsyncCommandResult::SUCCESS);
473 }
474
475 if (detectionDataCallback != nullptr) {
476 detectionDataCallback(this, detectionData.presenceDetected);
477 };
478 DEBUG_PRINTLN_DATA("DATA received");
479 return true;
480 }
481 DEBUG_PRINTLN_DATA("DATA invalid");
482 return false;
483}

717bool LD2410Async::executeCommandSequenceAsync(AsyncCommandCallback callback) {
718 if (sendCommandAsyncCommandPending || executeCommandSequencePending) return false;
719 if (commandSequenceBufferCount == 0) return true; // nothing to send
720
721 DEBUG_PRINT_MILLIS;
722 DEBUG_PRINT("Starting command sequence execution. Number of commands: ");
723 DEBUG_PRINTLN(commandSequenceBufferCount);
724
725 executeCommandSequencePending = true;
726 executeCommandSequenceCallback = callback;
727 executeCommandSequenceInitialConfigModeState = configModeEnabled;
728 executeCommandSequenceStartMs = millis();
729
730 if (commandSequenceBufferCount == 0) {
731 //Wait 1 ms to ensure that the callback is not executed before the caller of this method has finished its work
732 executeCommandSequenceOnceTicker.once_ms(1, [this]() {
733 this->executeCommandSequenceAsyncExecuteCallback(LD2410Async::AsyncCommandResult::SUCCESS);
734 });
735 return true;
736
737 }
738
739 if (!configModeEnabled) {
740 // Need to enable config mode first
741 // So set the sequence index to -1 to ensure the first command in the sequence (at index 0) is executed when the callback fires.
742 executeCommandSequenceIndex = -1;
743 return sendCommandAsync(LD2410Defs::configEnableCommandData, 1, executeCommandSequenceAsyncCommandCallback); //Retry once since this command will sometimes not work resp. not send an ack.
744 }
745 else {
746 // Already in config mode, start directly.
747 // We start the first command in the sequence directly and set the sequence index 0, so the second command (if any) is executed when the callback fires.
748 executeCommandSequenceIndex = 0;
749 return sendCommandAsync(commandSequenceBuffer[executeCommandSequenceIndex], executeCommandSequenceAsyncCommandCallback);

754bool LD2410Async::addCommandToSequence(const byte* command) {
755 if (sendCommandAsyncCommandPending || executeCommandSequencePending) return false;
756
757 if (commandSequenceBufferCount >= MAX_COMMAND_SEQUENCE_LENGTH) {
758 DEBUG_PRINT_MILLIS;
759 DEBUG_PRINTLN("Error: Command sequence buffer full.");
760 return false;
761 };
762 // First byte of the command is the payload length
763 uint8_t len = command[0];
764 uint8_t totalLen = len + 2; // payload + 2 length bytes
765
766 if (totalLen > LD2410Defs::LD2410_Buffer_Size) {
767 DEBUG_PRINT_MILLIS;
768 DEBUG_PRINTLN("Error: Command too long for command sequence buffer.");
769 return false; // safety
770 }
771 memcpy(commandSequenceBuffer[commandSequenceBufferCount],
772 command,
773 totalLen);
774
775 commandSequenceBufferCount++;
776 DEBUG_PRINT_MILLIS;
777 DEBUG_PRINT("Command added to sequence. Count: ");
778 DEBUG_PRINTLN(commandSequenceBufferCount);

1215bool LD2410Async::configureAllConfigSettingsAsyncBuildSaveChangesCommandSequence() {
1216 //Get a clone of the current config, so it does not get changed while we are working
1217 LD2410Types::ConfigData currentConfig = LD2410Async::getConfigData();
1218
1219 if (!resetCommandSequence()) {
1220 DEBUG_PRINT_MILLIS;
1221 DEBUG_PRINTLN("Error: Could not reset command sequence.");
1222 return false;
1223 };
1224 // 1. Max gate + no one timeout
1225 if (configureAllConfigSettingsAsyncWriteFullConfig
1226 || currentConfig.maxMotionDistanceGate != configureAllConfigSettingsAsyncConfigDataToWrite.maxMotionDistanceGate
1227 || currentConfig.maxStationaryDistanceGate != configureAllConfigSettingsAsyncConfigDataToWrite.maxStationaryDistanceGate
1228 || currentConfig.noOneTimeout != configureAllConfigSettingsAsyncConfigDataToWrite.noOneTimeout) {
1229 byte cmd[sizeof(LD2410Defs::maxGateCommandData)];
1230 if (!LD2410CommandBuilder::buildMaxGateCommand(cmd,
1231 configureAllConfigSettingsAsyncConfigDataToWrite.maxMotionDistanceGate,
1232 configureAllConfigSettingsAsyncConfigDataToWrite.maxStationaryDistanceGate,
1233 configureAllConfigSettingsAsyncConfigDataToWrite.noOneTimeout)) {
1234 DEBUG_PRINT_MILLIS;
1235 DEBUG_PRINTLN("Error: Building max gate command failed.");
1236 return false;
1237 };
1238 if (!addCommandToSequence(cmd)) {
1239 DEBUG_PRINT_MILLIS;
1240 DEBUG_PRINTLN("Error: Adding max gate command to sequence failed.");
1241 return false;
1242 };
1243 DEBUG_PRINT_MILLIS;
1244 DEBUG_PRINTLN("Max gate command added to sequence.");
1245 }
1246 // 2. Gate sensitivities (sequence of commands)
1247 for (byte gate = 0; gate < 9; gate++) {
1248 if (configureAllConfigSettingsAsyncWriteFullConfig
1249 || currentConfig.distanceGateMotionSensitivity[gate] != configureAllConfigSettingsAsyncConfigDataToWrite.distanceGateMotionSensitivity[gate]
1250 || currentConfig.distanceGateStationarySensitivity[gate] != configureAllConfigSettingsAsyncConfigDataToWrite.distanceGateStationarySensitivity[gate]) {
1251 byte cmd[sizeof(LD2410Defs::distanceGateSensitivityConfigCommandData)];
1252 if (!LD2410CommandBuilder::buildGateSensitivityCommand(cmd, gate,
1253 configureAllConfigSettingsAsyncConfigDataToWrite.distanceGateMotionSensitivity[gate],
1254 configureAllConfigSettingsAsyncConfigDataToWrite.distanceGateStationarySensitivity[gate])) {
1255 DEBUG_PRINT_MILLIS;
1256 DEBUG_PRINT("Error: Error building gate sensitivity command for gate ");
1257 DEBUG_PRINTLN(gate);
1258
1259 return false;
1260 };
1261 if (!addCommandToSequence(cmd)) {
1262 DEBUG_PRINT_MILLIS;
1263 DEBUG_PRINT("Error: Adding gate sensitivity command for gate ");
1264 DEBUG_PRINT(gate);
1265 DEBUG_PRINTLN(" to sequence failed.");
1266
1267 return false;
1268 }
1269 DEBUG_PRINT_MILLIS
1270 DEBUG_PRINT("Gate sensitivity command for gate ");
1271 DEBUG_PRINT(gate);
1272 DEBUG_PRINTLN(" added to sequence.");
1273 }
1274 }
1275
1276 //3. Distance resolution
1277 if (configureAllConfigSettingsAsyncWriteFullConfig
1278 || currentConfig.distanceResolution != configureAllConfigSettingsAsyncConfigDataToWrite.distanceResolution) {
1279 byte cmd[6]; // resolution commands are 6 bytes long
1280 if (!LD2410CommandBuilder::buildDistanceResolutionCommand(cmd, configureAllConfigSettingsAsyncConfigDataToWrite.distanceResolution)) {
1281 DEBUG_PRINT_MILLIS;
1282 DEBUG_PRINTLN("Error: Building distance resolution command failed.");
1283 return false;
1284 };
1285 if (!addCommandToSequence(cmd)) {
1286 DEBUG_PRINT_MILLIS;
1287 DEBUG_PRINTLN("Error: Adding distance resolution command to sequence failed.");
1288 return false;
1289 };
1290 DEBUG_PRINT_MILLIS;
1291 DEBUG_PRINTLN("Distance resolution command added to sequence.");
1292 };
1293
1294 //4. Aux control settings
1295 if (configureAllConfigSettingsAsyncWriteFullConfig
1296 || currentConfig.lightControl != configureAllConfigSettingsAsyncConfigDataToWrite.lightControl
1297 || currentConfig.lightThreshold != configureAllConfigSettingsAsyncConfigDataToWrite.lightThreshold
1298 || currentConfig.outputControl != configureAllConfigSettingsAsyncConfigDataToWrite.outputControl) {
1299 byte cmd[sizeof(LD2410Defs::setAuxControlSettingCommandData)];
1300 if (!LD2410CommandBuilder::buildAuxControlCommand(cmd,
1301 configureAllConfigSettingsAsyncConfigDataToWrite.lightControl,
1302 configureAllConfigSettingsAsyncConfigDataToWrite.lightThreshold,
1303 configureAllConfigSettingsAsyncConfigDataToWrite.outputControl)) {
1304 DEBUG_PRINT_MILLIS;
1305 DEBUG_PRINTLN("Error: Building aux control command failed.");
1306 return false;
1307 };
1308 if (!addCommandToSequence(cmd)) {
1309 DEBUG_PRINT_MILLIS;
1310 DEBUG_PRINTLN("Error: Adding aux control command to sequence failed.");
1311 return false;
1312 };
1313 DEBUG_PRINT_MILLIS;
1314 DEBUG_PRINTLN("Aux control command added to sequence.");
1315 };
1316 return true;
1317};

1620void LD2410Async::handleInactivity() {
1621
1622 if (inactivityHandlingEnabled && inactivityHandlingTimeoutMs > 0) {
1623 unsigned long timeoutToUse = inactivityHandlingTimeoutMs;
1624 if (timeoutToUse < asyncCommandTimeoutMs + 1000) {
1625 timeoutToUse = asyncCommandTimeoutMs + 1000;
1626 }
1627 unsigned long currentTime = millis();
1628 unsigned long inactiveDurationMs = currentTime - lastSensorActivityTimestamp;
1629 if (lastSensorActivityTimestamp != 0 && inactiveDurationMs > timeoutToUse) {
1630 if (inactivityHandlingStep == 0 || currentTime - lastInactivityHandlingTimestamp > asyncCommandTimeoutMs + 1000) {
1631 lastInactivityHandlingTimestamp = currentTime;
1632
1633 switch (inactivityHandlingStep++) {
1634 case 0:
1635 DEBUG_PRINT_MILLIS;
1636 DEBUG_PRINTLN("Inactivity handling cancels pending async operations");
1637 asyncCancel();
1638 break;
1639 case 1:
1640 DEBUG_PRINT_MILLIS;
1641 DEBUG_PRINTLN("Inactivity handling tries to force diable config mode");
1642 //We dont care about the result resp. a callback, if the command has the desired effect, fine, otherwise will try to reboot the sensor anyway
1643 disableConfigModeInternalAsync(true, nullptr);
1644 break;
1645 case 3:
1646 DEBUG_PRINT_MILLIS;
1647 DEBUG_PRINTLN("Inactivity handling reboots the sensor");
1648 //We dont care about the result, if the command has the desired effect, fine, otherwise will try to reboot the sensor anyway
1649 rebootAsync(true, nullptr);
1650
1651 break;
1652 default:
1653 DEBUG_PRINT_MILLIS;
1654 DEBUG_PRINTLN("Inactivity handling could not revert sensor to normal operation. Reset the inactivity timeout and try again after the configured inactivity period.");
1655 //Inactivity handling has tried everything it can do, so reset the inactivity handling steps and call heartbeat. This will ensure inactivity handling will get called again after the inactivity handling period
1656 inactivityHandlingStep = 0;
1657 heartbeat();
1658 break;
1659 }
1660 }
1661 }
1662 else {
1663 inactivityHandlingStep = 0;
1664 }
1665 }
