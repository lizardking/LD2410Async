3* @brief Example: Basic Presence Detection with LD2410Async
4*
5* @details
6* This sketch demonstrates how to use the LD2410Async library to detect presence
7* using only the presenceDetected variable from the detection data callback.
8* It prints a message with a timestamp whenever the presence state changes.
9*
10* @warning
11* Important!
12* Adjust RADAR_RX_PIN and RADAR_TX_PIN to match your wiring.

40// Track last presence state
41bool lastPresenceDetected = false;
42bool firstCallback = true;
43
44/**
45* @brief Callback function called whenever new detection data arrives
46*
47* @details
48* Since only basic presence information is required in the example, it is just using the presenceDetected para of the callback.
49* The logic in this methods just ensures that only changes in presence state are printed to the Serial Monitor.
50*
51 * @note
52 * If only basic presence detection is needed, use the presenceDetected variable directly instead of accessing the full detection data struct.
53 * For more advanced use cases, the full detection data can be accessed using getDetectionDataRef() or getDetectionData().
54*/
55void onDetectionDataReceived(LD2410Async* sender, bool presenceDetected, byte userData) {
56 if (firstCallback || presenceDetected != lastPresenceDetected) {
57 unsigned long now = millis();
58 Serial.print("[");
59 Serial.print(now);
60 Serial.print(" ms] Presence detected: ");
61 Serial.println(presenceDetected ? "YES" : "NO");
62 lastPresenceDetected = presenceDetected;
63 firstCallback = false;
64 }
65}

67/**
68* @brief Arduino setup function which initializes the radar and registers the callback
69*
70* @details
71* radar.begin() starts the background task of the LD2410Async library which automatically handles
72* incoming data and triggers callbacks. The onDetectionDataReceived callback is registered to receive detection data.
73*
74*/
75void setup() {
76 // Initialize USB serial for debug output
77 Serial.begin(115200);
78 while (!Serial) {
79 ; // wait for Serial Monitor
80 }
81 Serial.println("LD2410Async Example: Basic Presence Detection");
82
83 // Initialize Serial1 with user-defined pins and baudrate
84 RadarSerial.begin(RADAR_BAUDRATE, SERIAL_8N1, RADAR_RX_PIN, RADAR_TX_PIN);
85
86 // Start the radar background task (parses incoming data frames)
87 if (radar.begin()) {
88 Serial.println("Radar task started successfully.");
89 // Register callback for detection updates
90 radar.registerDetectionDataReceivedCallback(onDetectionDataReceived, 0);
91 }
92 else {
93 Serial.println("ERROR! Could not start radar task.");
94 }
95}
