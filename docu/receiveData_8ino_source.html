44/*
45 * @brief Callback function called whenever new detection data arrives
46 *
47 * @details
48 * The detection data is accessed via a reference to avoid making a copy. This is more efficient and saves memory.
49 * The callback provides the presenceDetected variable for convenience, but all other detection data is available
50 * in the DetectionData struct that can be accessed using .getDetectionDataRef()
51 *
52 * @note
53 * In callback methods is is generally advised to access members of the LD2410Async instance via the sender pointer.
54 * This ensures that you are allways working with the correct instance, which is important if you have multiple LD2410Async instances.
55 * Also keep in mind that callbacks should be as short and efficient as possible to avoid blocking the background task of the library.
56 */
57void onDetectionDataReceived(LD2410Async* sender, bool presenceDetected, byte userData) {
58 // Access detection data efficiently without making a copy
59 const LD2410Types::DetectionData& data = sender->getDetectionDataRef();
60
61 Serial.println("=== Detection Data ===");
62
63 Serial.print("Target State: ");
64 Serial.println(LD2410Types::targetStateToString(data.targetState));
65
66 Serial.print("Presence detected: ");
67 Serial.println(data.presenceDetected ? "Yes" : "No");
68
69 Serial.print("Moving presence: ");
70 Serial.println(data.movingPresenceDetected ? "Yes" : "No");
71
72 Serial.print("Stationary presence: ");
73 Serial.println(data.stationaryPresenceDetected ? "Yes" : "No");
74
75 Serial.print("Moving Distance (cm): ");
76 Serial.println(data.movingTargetDistance);
77
78 Serial.print("Stationary Distance (cm): ");
79 Serial.println(data.stationaryTargetDistance);
80
81 Serial.print("Light Level: ");
82 Serial.println(data.lightLevel);
83
84 Serial.println("======================");
85}

87/**
88* @brief Arduino setup function which initializes the radar and registers the callback
89*
90* @details
91* radar.begin() starts the background task of the LD2410Async library which automatically handles
92* incoming data and triggers callbacks. The onDetectionDataReceived callback is registered to receive detection data.
93*
94*/
95void setup() {
96 // Initialize USB serial for debug output
97 Serial.begin(115200);
98 while (!Serial) {
99 ; // wait for Serial Monitor
100 }
101 Serial.println("LD2410Async Example: Receive Data");
102
103 // Initialize Serial1 with user-defined pins and baudrate
104 RadarSerial.begin(RADAR_BAUDRATE, SERIAL_8N1, RADAR_RX_PIN, RADAR_TX_PIN);
105
106 // Start the radar background task (parses incoming data frames)
107 if (radar.begin()) {
108 Serial.println("Radar task started successfully.");
109 // Register callback for detection updates
110 radar.registerDetectionDataReceivedCallback(onDetectionDataReceived, 0);
111 }
112 else {
113 Serial.println("ERROR! Could not start radar task.");
114 }
