1/**
2 * @brief Example: Webservice for Presence Detection with LD2410Async and ESP Async WebServer
3 *
4 * @details
5 * This example demonstrates how to use the LD2410Async library with an ESP32 to provide a web-based
6 * presence detection service. The ESP32 connects to a WiFi network, starts a web server, and serves
7 * a simple HTML page. The page connects via WebSocket to receive live presence detection data
8 * (presenceDetected, detectedDistance) from the LD2410 radar sensor.
9 *
10 *
11 * Dependencies:
12 * - ESPAsyncWebServer: https://github.com/me-no-dev/ESPAsyncWebServer
13 * - AsyncTCP: https://github.com/me-no-dev/AsyncTCP
14 * - LD2410Async library
15 *
16 * @warning
17 * Make sure to install the required libraries and adjust the configuration section below
18 * to match your hardware and network setup.
19 */

92/**
93 * @brief Notifies all connected WebSocket clients with the latest presence data.
94 *
95 * Sends a JSON string with the fields "presenceDetected" (bool) and "detectedDistance" (uint16_t).
96 */
97void notifyClients() {
98 String msg = "{\"presenceDetected\":";
99 msg += (latestPresenceDetected ? "true" : "false");
100 msg += ",\"detectedDistance\":";
101 msg += latestDetectedDistance;
102 msg += "}";
103 ws.textAll(msg);
104}

106/**
107 * @brief Callback function called whenever new detection data arrives from the radar.
108 *
109 * @param sender Pointer to the LD2410Async instance (for multi-sensor setups)
110 * @param presenceDetected True if presence is detected, false otherwise
111 * @param userData User data (not used)
112 */
113void onDetectionDataReceived(LD2410Async* sender, bool presenceDetected, byte userData) {
114 const LD2410Types::DetectionData& data = sender->getDetectionDataRef();
115 latestPresenceDetected = data.presenceDetected;
116 latestDetectedDistance = data.detectedDistance;
117 notifyClients();
118}

120/**
121 * @brief Handles WebSocket events (client connect/disconnect).
122 *
123 * @param server Pointer to the AsyncWebSocket server
124 * @param client Pointer to the connected client
125 * @param type Event type (connect, disconnect, etc.)
126 * @param arg Event argument
127 * @param data Event data
128 * @param len Data length
129 */
130void onWsEvent(AsyncWebSocket* server, AsyncWebSocketClient* client, AwsEventType type,
131 void* arg, uint8_t* data, size_t len) {
132 if (type == WS_EVT_CONNECT) {
133 Serial.printf("WebSocket client #%u connected\n", client->id());
134 // Send initial state
135 notifyClients();
136 }
137 else if (type == WS_EVT_DISCONNECT) {
138 Serial.printf("WebSocket client #%u disconnected\n", client->id());
139 }

193/**
194 * @brief Connects to WiFi using the user configuration.
195 *
196 * If USE_STATIC_IP is true, configures a static IP, subnet, and gateway.
197 * Prints connection status and IP address to Serial.
198 */
199void setupWiFi() {
200 Serial.print("Connecting to WiFi: ");
201 Serial.println(WIFI_SSID);
202 WiFi.mode(WIFI_STA);
203#if USE_STATIC_IP
204 if (!WiFi.config(LOCAL_IP, GATEWAY, SUBNET)) {
205 Serial.println("STA Failed to configure static IP");
206 }
207#endif
208 WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
209 int retries = 0;
210 while (WiFi.status() != WL_CONNECTED) {
211 delay(500);
212 Serial.print(".");
213 if (++retries > 40) { // 20 seconds timeout
214 Serial.println("\nFailed to connect to WiFi!");
215 return;
216 }
217 }
218 Serial.println("\nWiFi connected.");
219 Serial.print("IP address: ");
220 Serial.println(WiFi.localIP());
221}

223/**
224 * @brief Arduino setup function.
225 *
226 * Initializes serial, WiFi, radar sensor, and webserver.
227 * Registers the detection data callback and sets up the WebSocket and HTTP handlers.
228 */
229void setup() {
230 // Serial for debug output
231 Serial.begin(115200);
232 while (!Serial) { ; }
233 Serial.println("LD2410Async Web Presence Detection Example");
234
235 // Setup WiFi
236 setupWiFi();
237
238 // Start UART for radar
239 RadarSerial.begin(RADAR_BAUDRATE, SERIAL_8N1, RADAR_RX_PIN, RADAR_TX_PIN);
240
241 // Start radar background task
242 if (radar.begin()) {
243 Serial.println("Radar task started successfully.");
244 radar.registerDetectionDataReceivedCallback(onDetectionDataReceived, 0);
245 }
246 else {
247 Serial.println("ERROR! Could not start radar task.");
248 }
249
250 // WebSocket setup
251 ws.onEvent(onWsEvent);
252 server.addHandler(&ws);
253
254 // Serve HTML page at "/"
255 server.on("/", HTTP_GET, [](AsyncWebServerRequest* request) {
256 request->send_P(200, "text/html", index_html);
257 });
258
259 // Start server
260 server.begin();
261 Serial.println("Webserver started.");
262}
